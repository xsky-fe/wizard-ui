将 Panel 用在元素上。

## 何时使用

- 增强文案显示

### 常规

```jsx
<div>
  <Panel>i am panel </Panel>
</div>
```

### 带颜色

```jsx
<div>
  <Panel bsStyle="warning">i am panel and warning style </Panel>
  <Panel bsStyle="warning" header="i am header">
    i have header and warning style
  </Panel>
  <Panel bg="primary" text="white">
    i am panel and have primary bg color
  </Panel>
  <Panel text="danger">i am panel and have danger text color </Panel>
</div>
```

### 带头部或底部

```jsx
<div>
  <Panel header={<p>i am panel header</p>}>i am panel</Panel>
  <Panel footer={<p>i am panel footer</p>}>i am panel</Panel>
</div>
```

### 带折叠

```jsx
<div>
  <Panel collapsible header={<p>i am panel header</p>}>
    i am panel and can collapsible(collapsible = true)
  </Panel>
  <Panel collapsible={false} header={<p>i am panel header</p>}>
    i am panel and can not collapsible(collapsible = false)
  </Panel>
  <Panel defaultExpanded header={<p>i am panel header</p>}>
    i am panel and default expand body
  </Panel>
  <Panel
    expanded={false}
    collapsible
    header={<p>i am panel and collapsible but can not expand</p>}
  >
    i am panel and collapsible but can not expanded
  </Panel>
  <Panel expanded collapsible header={<p>i am panel header</p>}>
    i am panel and and expanded but collapsible does not work
  </Panel>
  <Panel expanded header={<p>i am panel header</p>}>
    i am panel and expand body
  </Panel>
</div>
```

```jsx
() => {
  const [open, setCollapse] = React.useState(true);
  const handleSelect = () => setCollapse(!open);
  return (
    <div>
      <h3>
        <p onClick={handleSelect}>i am onClick can expand body</p>
      </h3>
      <Panel collapsible expanded={open}>
        i am collaspe
      </Panel>
    </div>
  );
};
```

```jsx
() => {
  const [open, setCollapse] = React.useState(true);
  const handleSelect = () => setCollapse(!open);
  return (
    <div>
      <Panel
        collapsible
        expanded={open}
        onSelect={handleSelect}
        header="I am select header"
      >
        i am collaspe
      </Panel>
    </div>
  );
};
```

### role 的

```jsx
<div>
  <Panel collapsible panelRole="collapse" header={<p>i am panel header</p>}>
    i am panel role in collapse
  </Panel>
  <Panel headerRole="collapse" header={<p>i am panel header</p>}>
    i am panel role in header
  </Panel>
</div>
```
